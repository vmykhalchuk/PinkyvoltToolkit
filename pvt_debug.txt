struct ArduinoPin {
    volatile uint8_t* port;
    volatile uint8_t* ddr;
    volatile uint8_t* pinReg;
    uint8_t bit;
};

---------- SOLUTION #3 ------------
template <uint8_t PIN>
class PinkyVoltOneWireErrorTransmitter final {
  private:
    static constexpr uint8_t bit = digitalPinToBitMask(PIN);
    static constexpr uint8_t port = digitalPinToPort(PIN);
    
    // Safety check that works across all ports
    static_assert(digitalPinToPort(PIN) != NOT_A_PIN, "Invalid Arduino Pin!");

    void inline setDPinToHigh() {
        volatile uint8_t *out = portOutputRegister(port);
        *out |= bit; 
        // Note: SBI assembly only works if the address is known at compile time. 
        // Modern GCC with optimizations often converts the above to an SBI anyway!
    }
};
